Модуль fs: Модуль fs (файловая система) в Node.js предоставляет API для взаимодействия с файловой системой. С помощью него можно выполнять различные операции с файлами, такие как чтение, запись, удаление, копирование, перемещение, создание папок, чтение и запись JSON-файлов, и многое другое. Модуль поддерживает как синхронные, так и асинхронные методы.

Асинхронные методы: Асинхронные методы используют колбэки для обработки данных после завершения операции. Ошибки обрабатываются в первом аргументе функции-колбэка.

Синхронные методы: Синхронные методы выполняются блокирующим образом, что означает, что выполнение следующего кода будет приостановлено до завершения текущей операции.

Модуль fs-extra: Расширяет стандартный модуль fs и добавляет больше возможностей, таких как копирование файлов, перемещение папок, удаление каталогов и более простой способ работы с JSON.

// Импорт модуля файловой системы
const fs = require('fs');

// Пример асинхронного чтения файла
fs.readFile('example.txt', 'utf8', (err, data) => {
    if (err) {
        // Обработка ошибки, если файл не найден или произошла другая ошибка
        console.error(err);
        return;
    }
    // Вывод содержимого файла в консоль
    console.log(data);
});

// Пример асинхронной записи в файл
const content = 'Hello, world!';
fs.writeFile('example.txt', content, err => {
    if (err) {
        // Обработка ошибки записи
        console.error(err);
        return;
    }
    // Уведомление об успешной записи
    console.log('File has been written');
});

// Пример синхронного чтения файла
try {
    const data = fs.readFileSync('example.txt', 'utf8');
    console.log(data); // Вывод содержимого файла
} catch (err) {
    // Обработка ошибки при чтении файла
    console.error(err);
}

// Пример синхронной записи в файл
const syncContent = 'Hello, world!';
try {
    fs.writeFileSync('example.txt', syncContent);
    console.log('File has been written'); // Уведомление об успешной записи
} catch (err) {
    // Обработка ошибки записи
    console.error(err);
}

// Пример асинхронного удаления файла
fs.unlink('example.txt', err => {
    if (err) {
        // Обработка ошибки удаления
        console.error(err);
        return;
    }
    // Уведомление об успешном удалении
    console.log('File has been deleted');
});

// Пример использования модуля fs-extra
const fse = require('fs-extra');

// Копирование файла
fse.copy('source.txt', 'destination.txt', err => {
    if (err) {
        console.error(err); // Обработка ошибки копирования
        return;
    }
    console.log('File has been copied'); // Уведомление об успешном копировании
});
